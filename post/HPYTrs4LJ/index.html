A. 巧克力（chocolates）

分析

40分做法：前缀和预处理，然后枚举，时间复杂度 O(n2)O(n^{2})O(n2)

正解有许多做法，在此介绍其中一种：双指针。左右两个指针，分别指向头尾，若此时 s1>s2s1 > s2s1>s2，右边加码，否则左边加码。时间复杂度 O(n)O(n)O(n)。

参考代码

#include<iostream>
#include<cstdio>

using namespace std;

const int N = 1e6 + 10;

long long n, a[N];
long long s1, s3, ans;

int main(){

    freopen("chocolates.in", "r", stdin);
    freopen("chocolates.out", "w", stdout);

    scanf("%d", &n);

    for(int i = 1; i <= n; i++){
        scanf("%lld", &a[i]);
    }

    for(int i = 0, j = n+1; i < j && i <= n; ){
        if(s1 == s3){
            ans = max(ans, s1);
            i++;
            j--;
            s1 += a[i];
            s3 += a[j];
        }else{
            if(s1 < s3){
                i++;
                s1 += a[i];
            }else{
                j--;
                s3 += a[j];
            }
        }
    }

    printf("%lld\n", ans);

    return 0;
}

B. 乐乐的城市（city）

分析

建树，然后怎么做都行。

一种做法是，从根节点 111 开始深搜，每个点的 disdisdis 等于它父亲节点加一。

参考代码

#include<iostream>
#include<cstdio>
#include<vector>

using namespace std;

const int N = 15;

int n, u, v;
int dis[N], ans;
vector <int> g[N];

void dfs(int x, int fa){
    for(int i = 0; i < g[x].size(); i++){
        if(g[x][i] != fa){
            dis[g[x][i]] = dis[x] + 1;
            dfs(g[x][i], x);
        }
    }
    ans = max(ans, dis[x]);
    return ;
}

int main(){
    
    freopen("city.in", "r", stdin);
    freopen("city.out", "w", stdout);
    
    scanf("%d", &n);
    
    for(int i = 1; i < n; i++){
        scanf("%d%d", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    
    dfs(1, 0);
    
    printf("%d\n", ans);
    
    return 0;
}

C. 表示（express）

分析

考虑先将给定的数转为三进制，则每一位有可能为 000 或 111 或 222。但根据题意，为222 是不合法的，只能为 000 或 111 或 −1-1−1。

由

2⋅3x=3x+1−3x2 \cdot 3^{x} = 3^{x+1} - 3^{x}2⋅3x=3x+1−3x

可知，若某一位为 222，则可将此位改为 −1-1−1，更高一位增加 111。

若出现某一位为 333，进位即可。

这题本来是作为思维题放在 A，但由于验题人反应难度不符，改到了 C 的位置。

注意 pow() 函数的精度问题！

参考代码

#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;

const int N = 40;

long long n, t = 1;
int ans[N];
int lt;

int main(){
    
    freopen("express.in", "r", stdin);
    freopen("express.out", "w", stdout);

    scanf("%lld",&n);

    for(int i = 0; n; i++){
        ans[i] = n % 3;
        n /= 3;
    }

    for(int i = 0; i < N; i++){
        if(ans[i] != 0){
            lt = i;
        }
        if(ans[i] == 3){
            ans[i+1]++;
            ans[i] = 0;
        }
        if(ans[i] == 2){
            ans[i] = -1;
            ans[i+1]++;
        }
    }
    
    for(int i = 0; i <= lt; i++, t *= 3){
        if(ans[i]){
            printf("%lld ", t * ans[i]);
        }
    }

    return 0;
}

            n /= 3;
        k *= 3;
    }
}

D. 乐乐的花瓶（vase）

分析

一种做法是二分答案，二分花瓶的宽度，转为判定。

则用两个单调队列分别维护区间最大、最小值，若某一时刻最大值减最小值大于等于 DDD，则可行。

参考代码

// 邓力炫
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
int n,d,ans;
bool vis;
deque <int> q1,q2;
struct c{
	int x,y;
};
c a[100005];
bool cmp(c x,c y){
	return x.x < y.x;
}
bool check(int x){
	while(!q1.empty()) q1.pop_front();//min
	while(!q2.empty()) q2.pop_front();//max 
	for(int i=1;i<=n;i++){
		while(!q1.empty()&&a[q1.back()].y>a[i].y) q1.pop_back();
		while(!q2.empty()&&a[q2.back()].y<a[i].y) q2.pop_back();
		while(!q1.empty()&&a[i].x-a[q1.front()].x>x) q1.pop_front();
		while(!q2.empty()&&a[i].x-a[q2.front()].x>x) q2.pop_front();
		q1.push_back(i);
		q2.push_back(i);
		if(a[q2.front()].y-a[q1.front()].y>=d) return 1; 
	}
	return 0;
}
int main(){
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout); 
	scanf("%d%d",&n,&d);
	for(int i=1;i<=n;i++) scanf("%d%d",&a[i].x,&a[i].y);
	sort(a+1,a+n+1,cmp);
	int l = 0,r = 1e6+5,mid;
	while(l<=r){
		mid = (l+r)/2;
		if(check(mid)){
			vis = 1;
			ans = mid;
			r = mid-1;
		}
		else l = mid+1;
	}
	if(vis) printf("%d\n",ans);
	else printf("-1\n");
	return 0;
}

Ex.1 A + B Problem（add）

分析

四则运算符都被禁了，只能考虑位运算。

我们知道，在二进制下，「异或（^、⊕\oplus⊕）」相当于不进位加法，而「与运算（&）」又能求出所有进位。

则 a+b=(a⊕b)+((a&b)<<1)a + b = (a \oplus b) + ((a \& b) << 1)a+b=(a⊕b)+((a&b)<<1)。

这样就用位运算把 a+ba + ba+b 转化为另两个数相加，不断重复，直到没有进位。

要开 long long ！

参考代码

#include<iostream>
#include<cstdio>

using namespace std;

unsigned long long a, b, c;

int main(){

    freopen("add.in", "r", stdin);
    freopen("add.out", "w", stdout);

    scanf("%llu%llu", &a, &b);

    while(b){
        c = a ^ b;
        b = (a & b) << 1;
        a = c;
    }

    printf("%llu", a);

    return 0;
}

赛时jzzzzzp给出了递归的写法：

#include "stdio.h"
long long a, b;
long long add(long long a, long long b) {
    if (!a || !b)
        return a | b;
    return add(a ^ b, (a & b) << 1);
}
main() {
    freopen("add.in", "r", stdin);
    freopen("add.out", "w", stdout);
    scanf("%lld%lld", &a, &b);
    printf("%lld\n", add(a, b));
}
